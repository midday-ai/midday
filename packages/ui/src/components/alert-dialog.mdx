  AlertDialog,
  AlertDialogAction,
  AlertDialogCancel,
  AlertDialogContent,
  AlertDialogDescription,
  AlertDialogFooter,
  AlertDialogHeader,
  AlertDialogTitle,
  AlertDialogTrigger,
} from "@/primitives/alert-dialog";`;

export const usageCode = `<AlertDialog>

  </AlertDialogContent>
</AlertDialog>`;
